function loadBible(version) { const contentDiv = document.getElementById('bible-content'); contentDiv.innerHTML = '

데이터를 불러오는 중입니다...

'; // 1. URL이 정확한지 다시 확인 (대소문자 구분 주의) const url = `https://raw.githubusercontent.com{version}.json`; fetch(url) .then(response => { if (!response.ok) throw new Error('파일을 찾을 수 없습니다: ' + response.status); return response.json(); }) .then(data => { contentDiv.innerHTML = ''; // 2. 데이터 구조 확인 (books가 없는 경우 예외 처리) if (!data.books) { contentDiv.innerHTML = '

데이터 구조가 올바르지 않습니다. (books 필드 없음)

'; return; } const books = data.books; for (const bookName in books) { const bookHeader = document.createElement('h2'); bookHeader.className = 'book-title'; bookHeader.innerText = bookName; contentDiv.appendChild(bookHeader); const chapters = books[bookName]; for (const chapterNum in chapters) { const chapterHeader = document.createElement('h3'); chapterHeader.className = 'chapter-title'; chapterHeader.innerText = `${chapterNum}장`; contentDiv.appendChild(chapterHeader); const verses = chapters[chapterNum]; for (const verseNum in verses) { const versePara = document.createElement('div'); versePara.className = 'verse'; // 3. 텍스트가 비어있을 경우에 대비 const text = verses[verseNum] || "내용이 없습니다."; versePara.innerHTML = `${verseNum}${text}`; contentDiv.appendChild(versePara); } } } }) .catch(error => { console.error('Error Details:', error); contentDiv.innerHTML = `

오류: ${error.message}

`; }); }
